              #include <iostream>
        #include <string>
        using namespace std;
         
        int main() {
          while (true) {
            string num;
            getline(cin, num);
         
            if (num == "0") break;
         
            long long res = 0;
         
            for (int i = 0; i < num.length(); i++) {
              int n = num[i] - '0';
              if (n > 4) n--;
              res = res * 9 + n;
            }
         
            cout << num << ": ";
            cout << res << endl;
          }
          return 0;
        }
