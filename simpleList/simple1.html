              #include <iostream>
        using namespace std;
         
        int main() {
          int n;
          cin >> n;
          
          int i = 2;
          int res = 1;
          while (n > 1) {
            int count = 0;
            while (n % i == 0) {
              n = n / i;
              count++;
            }
            res *= count * 2 + 1;
            i++;
          }
         
          cout << res << endl;
          return 0;
        }
